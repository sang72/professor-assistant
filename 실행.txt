Git에 커밋하기 (선택사항)

  # 강의안 추가 확인
  git status

  # 모든 강의안 스테이징
  git add courses/frm401_franchise_management_2026_fall/lectures/

  # 커밋
  git commit -m "Add complete FRM401 lecture notes (Week 1-15, 45 sessions) with Korean translation and latest 2024-2026
   cases"

  # 푸시
  git push origin main

  3️⃣ 백업 (권장)

  # 강의안 폴더 백업
  cp -r courses/frm401_franchise_management_2026_fall/lectures/ \
        ~/backups/frm401_lectures_2026_05_01/
